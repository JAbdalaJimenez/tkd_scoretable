function playBeep(freq=880, dur=0.1) {
  try {
    const ctx = new (window.AudioContext || window.webkitAudioContext)();
    const osc = ctx.createOscillator();
    const gain = ctx.createGain();
    osc.connect(gain);
    gain.connect(ctx.destination);
    osc.frequency.value = freq;
    gain.gain.setValueAtTime(0.3, ctx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + dur);
    osc.start();
    osc.stop(ctx.currentTime + dur);
  } catch(e) {}
}

function addScore(side, points, remote=false) {
  if (!remote) playBeep(points>0?660:300,0.08);
  if(side==='blue') S.scoreBlue = Math.max(0, S.scoreBlue + points);
  else S.scoreRed = Math.max(0, S.scoreRed + points);
  renderUI();
}

function addWarning(side, remote=false) {
  if (!remote) playBeep(440,0.15);
  if(side==='blue') {
    S.wBlue++;
    if(S.wBlue%2===0) addScore('red',1,remote);
  } else {
    S.wRed++;
    if(S.wRed%2===0) addScore('blue',1,remote);
  }
  renderUI();
}

function addYC(side, remote=false) {
  if(!remote) playBeep(330,0.2);
  if(side==='blue') S.ycBlue = Math.min(9, S.ycBlue+1);
  else S.ycRed = Math.min(9, S.ycRed+1);
  renderUI();
}

function toggleTimer(remote=false) {
  if(S.timerMs<=0) return;
  S.running = !S.running;
  if(S.running) {
    lastTimestamp = Date.now();
    timerInterval = setInterval(() => {
      const now = Date.now();
      S.timerMs = Math.max(0, S.timerMs - (now - lastTimestamp));
      lastTimestamp = now;
      renderUI();
      if(S.timerMs<=0) { clearInterval(timerInterval); S.running=false; renderUI(); }
    }, 50);
  } else clearInterval(timerInterval);
  renderUI();
}

function resetTimer(remote=false) {
  clearInterval(timerInterval);
  S.running = false;
  S.timerMs = 60000;
  renderUI();
}

function nextRound(remote=false) {
  if(S.round < S.maxRounds) { S.round++; resetTimer(); renderUI(); }
}

function resetAll(remote=false) {
  clearInterval(timerInterval);
  S.scoreBlue=0; S.scoreRed=0; S.wBlue=0; S.wRed=0; S.ycBlue=0; S.ycRed=0;
  S.round=1; S.timerMs=60000; S.running=false;
  renderUI();
}

  feed.prepend(item);

}
